# 链表的创建

创建一个长度为n的链表：

```c++
typedef struct HZ
{
  int data;
  HZ *next;
}*node,Node;
node creat(int n)
{
    node h = (node)malloc(sizeof(Node));
    h->next = NULL;
    node q = h;
    node p;
    for (int i = 0;i < n;i++)
    {
        p = (node)malloc(sizeof(Node));
        cin >> p->data;
        p->next = q->next;
        q->next = p;//前插法
    }
    return h;
}
```

# 链表的打印

```c++
void print(node h)
{
    if (h->next == NULL)
    {
        cout << "Link list is empty" << endl;
        return;
    }
    node q = h->next;
    while (q != NULL)
    {
        cout << q->data << " ";
        q = q->next;
    }
    cout << endl;
}
```

# 链表节点的删除

```c++
node del(node h,int a)//删除第a个节点
{
    node q = h;
    for (int i = 0;i < a - 1;i++)
    {
        q = q->next;
        if (q == NULL)
        {
            cout << "delete fail" << endl;
            return h;
        }
    }
    if (q->next == NULL)
    {
        cout << "delete fail" << endl;
    }
    else
    {
        node p = q->next;
        q->next = p->next;
        free(p);
        cout << "delete OK" << endl;
    }
    return h;
}
```

# 链表节点的插入

```c++
node insert(node h,int a,int e)//在第a个节点前面插入数据为e的节点
{
    node q = h;
    for (int i = 0;i < a - 1;i++)
    {
        q = q->next;
        if (q == NULL)
        {
            cout << "insert fail" << endl;
            return h;
        }
    }
    node p = (node)malloc(sizeof(Node));
    p->data = e;
    p->next = q->next;
    q->next = p;
    cout << "insert OK" << endl;
    return h;
}
```
